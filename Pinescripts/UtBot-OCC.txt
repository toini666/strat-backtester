grp_main = "Configuration de la Stratégie"
i_stratName = input.string("OCCUTBotEMA", "Nom de la stratégie", group=grp_main)

grp_occ = "OCC (Open Close Cross)"
i_useRes    = input.bool(true, "Utiliser Résolution Alternative ?", group=grp_occ)
i_intRes    = input.int(7, "Multiplicateur Résolution Alternative", minval=1, group=grp_occ)
i_basisType = input.string("SMMA", "Type MA", options=["SMA", "EMA", "DEMA", "TEMA", "WMA", "VWMA", "SMMA", "HullMA", "LSMA", "ALMA", "SSMA", "TMA"], group=grp_occ)
i_basisLen  = input.int(8, "Période MA", minval=1, group=grp_occ)
i_offsetSigma = input.int(6, "Offset LSMA / Sigma ALMA", minval=0, group=grp_occ)
i_offsetALMA  = input.float(0.85, "Offset ALMA", minval=0.0, step=0.01, group=grp_occ)
i_delayOffset = input.int(0, "Delay Open/Close MA (Non-Repainting)", minval=0, step=1, group=grp_occ)

grp_utbot = "UT Bot"
i_keyValue  = input.float(1.0, "Key Value (sensibilité)", group=grp_utbot)
i_atrPeriod = input.int(10, "Période ATR", group=grp_utbot)

grp_ema = "EMA Sortie Finale"
i_emaPeriod = input.int(20, "Période EMA", minval=1, group=grp_ema)

grp_risk = "Gestion des Risques"
i_tickBuffer    = input.int(2, "Buffer SL (en ticks)", group=grp_risk, tooltip="Marge de sécurité ajoutée sous/sur le SL (ex: 2 ticks = SL reculé de 2×mintick)")
i_rrPartial     = input.float(2.0, "Risk/Reward TP Partiel", group=grp_risk)
i_lookbackEntry = input.int(3, "Bougies max pour confirmation entrée", minval=1, group=grp_risk)
i_slLookback    = input.int(5, "Nombre de bougies pour le SL", minval=2, group=grp_risk, tooltip="Nombre de bougies (incluant la bougie setup) pour calculer le lowest/highest du SL")

grp_time = "Filtrage Horaire (Bruxelles)"
i_forceCloseHour = input.int(22, "Heure de clôture forcée", group=grp_time, tooltip="Toute position ouverte sera fermée à cette heure (Brussels)")

// ============================================
// VARIABLES D'ÉTAT (POSITION MANAGEMENT)
// ============================================
var bool positionActive  = false
var int  currentSide     = 0      // 1 = Long, -1 = Short
var float entryPrice     = na
var float stopPrice      = na
var float tpPrice        = na
var int  lastActionBar   = -1     // Sécurité same-bar

// État du setup en attente d'entrée
var bool  setupPending   = false
var int   setupSide      = 0      // 1 = long setup, -1 = short setup
var int   setupBar       = -1     // bar_index de la bougie du signal UT Bot
var int   setupBarsLeft  = 0      // compteur de bougies restantes pour la confirmation
var float setupStopLong  = na     // SL pré-calculé au moment du setup (long)
var float setupStopShort = na     // SL pré-calculé au moment du setup (short)

// État TP partiel
var bool  partialTaken   = false

// Pointeurs vers les lignes visuelles
var line lineEntry = na
var line lineStop  = na
var line lineTP    = na

// Variables de visualisation des sorties (pour plotshape au scope global)
var bool showSLLong      = false
var bool showSLShort     = false
var bool showBELong      = false
var bool showBEShort     = false
var bool showTPLong      = false
var bool showTPShort     = false
var bool showFinalLong   = false
var bool showFinalShort  = false
var bool showForceLong   = false
var bool showForceShort  = false

// Reset tous les flags visuels chaque bougie
showSLLong     := false
showSLShort    := false
showBELong     := false
showBEShort    := false
showTPLong     := false
showTPShort    := false
showFinalLong  := false
showFinalShort := false
showForceLong  := false
showForceShort := false

// ============================================
// FONCTIONS UTILITAIRES
// ============================================
roundToTick(float price) =>
    math.round(price / syminfo.mintick) * syminfo.mintick

// --- FILTRAGE HORAIRE (Timezone Bruxelles) ---
isBlackoutWindow() =>
    int hh = hour(time, "Europe/Brussels")
    int mm = minute(time, "Europe/Brussels")
    bool b1 = (hh == 0  and mm >= 0 and mm < 5)
    bool b2 = (hh == 9  and mm >= 0 and mm < 5)
    bool b3 = (hh == 15 and mm >= 30 and mm < 35)
    b1 or b2 or b3

isForceCloseTime() =>
    int hh = hour(time, "Europe/Brussels")
    hh >= i_forceCloseHour

bool inBlackout = isBlackoutWindow()
bool forceClose = isForceCloseTime()

// Génération de l'alerte JSON standardisée
// freqType: true = once_per_bar (immédiat), false = once_per_bar_close (confirmation)
triggerAlert(string type, string alertSide, float alertEntry, float alertStop, float alertTp, bool freqImmediate) =>
    string json = '{\n'
    json += '  "ticker": "' + syminfo.ticker + '",\n'
    json += '  "type": "' + type + '",\n'
    json += '  "side": "' + alertSide + '",\n'

    if type == "PARTIAL" or type == "CLOSE"
        if not na(alertEntry)
            json += '  "entry": ' + str.tostring(alertEntry) + ',\n'
    else if type == "SIGNAL"
        if not na(alertEntry)
            json += '  "entry": ' + str.tostring(alertEntry) + ',\n'
        if not na(alertStop)
            json += '  "stop": ' + str.tostring(alertStop) + ',\n'
        if not na(alertTp)
            json += '  "tp": ' + str.tostring(alertTp) + ',\n'
    else if type == "SETUP"
        if not na(alertEntry)
            json += '  "entry": ' + str.tostring(alertEntry) + ',\n'
        if not na(alertStop)
            json += '  "stop": ' + str.tostring(alertStop) + ',\n'
        if not na(alertTp)
            json += '  "tp": ' + str.tostring(alertTp) + ',\n'

    json += '  "strat": "' + i_stratName + '-' + syminfo.ticker + '-' + timeframe.period + '",\n'
    json += '  "timeframe": "' + timeframe.period + '"\n'
    json += '}'

    if freqImmediate
        alert(json, alert.freq_once_per_bar)
    else
        alert(json, alert.freq_once_per_bar_close)

// ============================================
// OCC (Open Close Cross) — Converti de v3 à v6
// ============================================
stratRes = timeframe.ismonthly ? str.tostring(timeframe.multiplier * i_intRes) + "M" : timeframe.isweekly ? str.tostring(timeframe.multiplier * i_intRes) + "W" : timeframe.isdaily ? str.tostring(timeframe.multiplier * i_intRes) + "D" : timeframe.isintraday ? str.tostring(timeframe.multiplier * i_intRes) : "60"

variant(string maType, float src, int len, int offSig, float offALMA) =>
    float v1 = ta.sma(src, len)
    float v2 = ta.ema(src, len)
    float v3 = 2 * v2 - ta.ema(v2, len)
    float v4 = 3 * (v2 - ta.ema(v2, len)) + ta.ema(ta.ema(v2, len), len)
    float v5 = ta.wma(src, len)
    float v6 = ta.vwma(src, len)
    float v7 = 0.0
    v7 := na(v7[1]) ? ta.sma(src, len) : (v7[1] * (len - 1) + src) / len
    float v8 = ta.wma(2 * ta.wma(src, len / 2) - ta.wma(src, len), math.round(math.sqrt(len)))
    float v9 = ta.linreg(src, len, offSig)
    float v10 = ta.alma(src, len, offALMA, offSig)
    float v11 = ta.sma(v1, len)
    float a1 = math.exp(-1.414 * 3.14159 / len)
    float b1 = 2 * a1 * math.cos(1.414 * 3.14159 / len)
    float c2 = b1
    float c3 = (-a1) * a1
    float c1 = 1 - c2 - c3
    float v12 = 0.0
    v12 := c1 * (src + nz(src[1])) / 2 + c2 * nz(v12[1]) + c3 * nz(v12[2])
    maType == "EMA" ? v2 : maType == "DEMA" ? v3 : maType == "TEMA" ? v4 : maType == "WMA" ? v5 : maType == "VWMA" ? v6 : maType == "SMMA" ? v7 : maType == "HullMA" ? v8 : maType == "LSMA" ? v9 : maType == "ALMA" ? v10 : maType == "TMA" ? v11 : maType == "SSMA" ? v12 : v1

float closeSeries = variant(i_basisType, close[i_delayOffset], i_basisLen, i_offsetSigma, i_offsetALMA)
float openSeries  = variant(i_basisType, open[i_delayOffset], i_basisLen, i_offsetSigma, i_offsetALMA)

float closeSeriesAlt = i_useRes ? request.security(syminfo.tickerid, stratRes, closeSeries, gaps=barmerge.gaps_off, lookahead=barmerge.lookahead_on) : closeSeries
float openSeriesAlt  = i_useRes ? request.security(syminfo.tickerid, stratRes, openSeries, gaps=barmerge.gaps_off, lookahead=barmerge.lookahead_on) : openSeries

float occUpper = math.max(closeSeriesAlt, openSeriesAlt)
float occLower = math.min(closeSeriesAlt, openSeriesAlt)

color trendColour = closeSeriesAlt > openSeriesAlt ? color.green : color.red
pClose = plot(closeSeriesAlt, title="OCC Close", color=trendColour, linewidth=2)
pOpen  = plot(openSeriesAlt, title="OCC Open", color=trendColour, linewidth=2)
fill(pClose, pOpen, color=color.new(trendColour, 80))

// ============================================
// UT BOT — Converti de v4 à v6
// ============================================
float xATR  = ta.atr(i_atrPeriod)
float nLoss = i_keyValue * xATR

float utSrc = close

float xATRTrailingStop = 0.0
xATRTrailingStop := if utSrc > nz(xATRTrailingStop[1], 0) and utSrc[1] > nz(xATRTrailingStop[1], 0)
    math.max(nz(xATRTrailingStop[1]), utSrc - nLoss)
else if utSrc < nz(xATRTrailingStop[1], 0) and utSrc[1] < nz(xATRTrailingStop[1], 0)
    math.min(nz(xATRTrailingStop[1]), utSrc + nLoss)
else if utSrc > nz(xATRTrailingStop[1], 0)
    utSrc - nLoss
else
    utSrc + nLoss

int utPos = 0
utPos := if utSrc[1] < nz(xATRTrailingStop[1], 0) and utSrc > nz(xATRTrailingStop[1], 0)
    1
else if utSrc[1] > nz(xATRTrailingStop[1], 0) and utSrc < nz(xATRTrailingStop[1], 0)
    -1
else
    nz(utPos[1], 0)

float utEma = ta.ema(utSrc, 1)
bool utAbove = ta.crossover(utEma, xATRTrailingStop)
bool utBelow = ta.crossover(xATRTrailingStop, utEma)

bool utBuy  = utSrc > xATRTrailingStop and utAbove
bool utSell = utSrc < xATRTrailingStop and utBelow

// ============================================
// EMA SORTIE FINALE
// ============================================
float emaExit = ta.ema(close, i_emaPeriod)
plot(emaExit, title="EMA Exit", color=color.orange, linewidth=1)

// ============================================
// LOGIQUE DE LA STRATÉGIE
// ============================================

// --- PRÉ-CALCUL DES SÉRIES (scope global, pas dans un if) ---
// lowest/highest des 5 dernières bougies (incluant la bougie courante)
float _lowestFiveLow   = ta.lowest(low, i_slLookback)
float _highestFiveHigh = ta.highest(high, i_slLookback)

// --- SETUP DETECTION ---
// Long Setup : UT Bot buy + bougie TOUCHE la bande OCC (low <= occUpper) + clôture AU-DESSUS (close > occUpper)
bool longSetupSignal  = utBuy and low <= occUpper and close > occUpper
// Short Setup : UT Bot sell + bougie TOUCHE la bande OCC (high >= occLower) + clôture EN-DESSOUS (close < occLower)
bool shortSetupSignal = utSell and high >= occLower and close < occLower

// --- 1. NOUVEAU SETUP (exécuté en premier, avant le timer) ---
// Nouveau setup long — SL = lowest(low,5) PRÉ-CALCULÉ au scope global, capturé sur la bougie du setup
if longSetupSignal and not positionActive and not inBlackout and not forceClose
    setupPending   := true
    setupSide      := 1
    setupBar       := bar_index
    setupBarsLeft  := i_lookbackEntry
    setupStopLong  := _lowestFiveLow - (i_tickBuffer * syminfo.mintick)

// Nouveau setup short — SL = highest(high,5) PRÉ-CALCULÉ au scope global
if shortSetupSignal and not positionActive and not inBlackout and not forceClose
    setupPending   := true
    setupSide      := -1
    setupBar       := bar_index
    setupBarsLeft  := i_lookbackEntry
    setupStopShort := _highestFiveHigh + (i_tickBuffer * syminfo.mintick)

// --- 2. CONFIRMATION ENTRÉE (exécutée AVANT le timer pour ne pas perdre 1 bougie) ---
// Long : le prix revient toucher la bande OCC (low <= occUpper) dans les N bougies suivantes
bool longEntryConfirm  = setupPending and setupSide == 1 and bar_index > setupBar and low <= occUpper
// Short : le prix revient toucher la bande OCC (high >= occLower) dans les N bougies suivantes
bool shortEntryConfirm = setupPending and setupSide == -1 and bar_index > setupBar and high >= occLower

// --- 3. TIMER (décrémente APRÈS le check d'entrée) ---
if setupPending and bar_index > setupBar and not longEntryConfirm and not shortEntryConfirm
    setupBarsLeft := setupBarsLeft - 1
    if setupBarsLeft <= 0
        setupPending   := false
        setupSide      := 0
        setupBar       := -1
        setupBarsLeft  := 0
        setupStopLong  := na
        setupStopShort := na

// --- CALCUL DES NIVEAUX D'ENTRÉE ---
// Prix d'entrée = le niveau de la bande OCC (pas le close !)
// Long: on entre au niveau occUpper (là où on touche la bande)
// Short: on entre au niveau occLower
float calcEntryLong  = occUpper
float calcEntryShort = occLower

// SL = pré-calculé et stocké au moment du setup
float calcStopLong  = setupStopLong
float calcStopShort = setupStopShort

float riskL = calcEntryLong - nz(calcStopLong, calcEntryLong)
float calcTpLong = calcEntryLong + (riskL * i_rrPartial)

float riskS = nz(calcStopShort, calcEntryShort) - calcEntryShort
float calcTpShort = calcEntryShort - (riskS * i_rrPartial)

// ============================================
// GESTION ET EXÉCUTION DES POSITIONS
// ============================================

// 0. CLÔTURE FORCÉE À 22H BRUXELLES (pas d'alerte, uniquement visuel)
if positionActive and forceClose and bar_index > lastActionBar
    if currentSide == 1
        showForceLong := true
    else
        showForceShort := true

    positionActive := false
    currentSide    := 0
    partialTaken   := false
    lastActionBar  := bar_index
    if not na(lineEntry)
        line.delete(lineEntry)
        lineEntry := na
    if not na(lineStop)
        line.delete(lineStop)
        lineStop := na
    if not na(lineTP)
        line.delete(lineTP)
        lineTP := na

// 1. ENTRÉE LONGUE (confirmation du setup) — signal once_per_bar (immédiat)
if not positionActive and longEntryConfirm and bar_index > lastActionBar and not inBlackout and not forceClose
    positionActive := true
    currentSide    := 1
    entryPrice     := roundToTick(calcEntryLong)
    stopPrice      := roundToTick(calcStopLong)
    tpPrice        := roundToTick(calcTpLong)
    partialTaken   := false
    lastActionBar  := bar_index

    setupPending   := false
    setupSide      := 0
    setupBar       := -1
    setupBarsLeft  := 0
    setupStopLong  := na

    lineEntry := line.new(bar_index, entryPrice, bar_index + 1, entryPrice, color=color.blue, style=line.style_dotted, width=1)
    lineStop  := line.new(bar_index, stopPrice, bar_index + 1, stopPrice, color=color.red, style=line.style_dotted, width=1)
    lineTP    := line.new(bar_index, tpPrice, bar_index + 1, tpPrice, color=color.green, style=line.style_dotted, width=1)
    triggerAlert("SIGNAL", "BUY", entryPrice, stopPrice, tpPrice, true)

// 2. ENTRÉE COURTE (confirmation du setup) — signal once_per_bar (immédiat)
else if not positionActive and shortEntryConfirm and bar_index > lastActionBar and not inBlackout and not forceClose
    positionActive := true
    currentSide    := -1
    entryPrice     := roundToTick(calcEntryShort)
    stopPrice      := roundToTick(calcStopShort)
    tpPrice        := roundToTick(calcTpShort)
    partialTaken   := false
    lastActionBar  := bar_index

    setupPending   := false
    setupSide      := 0
    setupBar       := -1
    setupBarsLeft  := 0
    setupStopShort := na

    lineEntry := line.new(bar_index, entryPrice, bar_index + 1, entryPrice, color=color.blue, style=line.style_dotted, width=1)
    lineStop  := line.new(bar_index, stopPrice, bar_index + 1, stopPrice, color=color.red, style=line.style_dotted, width=1)
    lineTP    := line.new(bar_index, tpPrice, bar_index + 1, tpPrice, color=color.green, style=line.style_dotted, width=1)
    triggerAlert("SIGNAL", "SELL", entryPrice, stopPrice, tpPrice, true)

// 3. SORTIES DE POSITIONS ACTIVES
// IMPORTANT: bar_index > lastActionBar empêche toute sortie (SL/TP) sur la bougie d'entrée
if positionActive and bar_index > lastActionBar
    // --- STOP LOSS ---
    bool hitSLLong  = currentSide == 1 and low <= stopPrice
    bool hitSLShort = currentSide == -1 and high >= stopPrice

    // --- TP PARTIEL ---
    bool hitTPLong  = currentSide == 1 and not partialTaken and high >= tpPrice
    bool hitTPShort = currentSide == -1 and not partialTaken and low <= tpPrice

    // --- TP FINAL (clôture de l'autre côté de l'EMA, uniquement après TP partiel) ---
    bool hitFinalLong  = currentSide == 1 and partialTaken and close < emaExit
    bool hitFinalShort = currentSide == -1 and partialTaken and close > emaExit

    // Priorité : SL > TP Partiel > TP Final
    if hitSLLong
        string closeSideStr = "SELL"
        float slExitPrice = stopPrice
        triggerAlert("CLOSE", closeSideStr, roundToTick(slExitPrice), na, na, false)

        if partialTaken
            showBELong := true
        else
            showSLLong := true

        positionActive := false
        currentSide    := 0
        partialTaken   := false
        lastActionBar  := bar_index
        if not na(lineEntry)
            line.delete(lineEntry)
            lineEntry := na
        if not na(lineStop)
            line.delete(lineStop)
            lineStop := na
        if not na(lineTP)
            line.delete(lineTP)
            lineTP := na

    else if hitSLShort
        string closeSideStr = "BUY"
        float slExitPrice = stopPrice
        triggerAlert("CLOSE", closeSideStr, roundToTick(slExitPrice), na, na, false)

        if partialTaken
            showBEShort := true
        else
            showSLShort := true

        positionActive := false
        currentSide    := 0
        partialTaken   := false
        lastActionBar  := bar_index
        if not na(lineEntry)
            line.delete(lineEntry)
            lineEntry := na
        if not na(lineStop)
            line.delete(lineStop)
            lineStop := na
        if not na(lineTP)
            line.delete(lineTP)
            lineTP := na

    else if hitTPLong
        partialTaken := true
        triggerAlert("PARTIAL", "SELL", roundToTick(tpPrice), na, na, true)

        stopPrice := entryPrice
        lastActionBar := bar_index

        if not na(lineTP)
            line.delete(lineTP)
            lineTP := na
        if not na(lineStop)
            line.set_xy1(lineStop, bar_index, stopPrice)
            line.set_xy2(lineStop, bar_index + 1, stopPrice)

        showTPLong := true

    else if hitTPShort
        partialTaken := true
        triggerAlert("PARTIAL", "BUY", roundToTick(tpPrice), na, na, true)

        stopPrice := entryPrice
        lastActionBar := bar_index

        if not na(lineTP)
            line.delete(lineTP)
            lineTP := na
        if not na(lineStop)
            line.set_xy1(lineStop, bar_index, stopPrice)
            line.set_xy2(lineStop, bar_index + 1, stopPrice)

        showTPShort := true

    else if hitFinalLong
        triggerAlert("CLOSE", "SELL", roundToTick(close), na, na, false)

        positionActive := false
        currentSide    := 0
        partialTaken   := false
        lastActionBar  := bar_index
        if not na(lineEntry)
            line.delete(lineEntry)
            lineEntry := na
        if not na(lineStop)
            line.delete(lineStop)
            lineStop := na
        if not na(lineTP)
            line.delete(lineTP)
            lineTP := na

        showFinalLong := true

    else if hitFinalShort
        triggerAlert("CLOSE", "BUY", roundToTick(close), na, na, false)

        positionActive := false
        currentSide    := 0
        partialTaken   := false
        lastActionBar  := bar_index
        if not na(lineEntry)
            line.delete(lineEntry)
            lineEntry := na
        if not na(lineStop)
            line.delete(lineStop)
            lineStop := na
        if not na(lineTP)
            line.delete(lineTP)
            lineTP := na

        showFinalShort := true

    else
        if not na(lineEntry)
            line.set_xy1(lineEntry, bar_index, entryPrice)
            line.set_xy2(lineEntry, bar_index + 1, entryPrice)
        if not na(lineStop)
            line.set_xy1(lineStop, bar_index, stopPrice)
            line.set_xy2(lineStop, bar_index + 1, stopPrice)
        if not na(lineTP)
            line.set_xy1(lineTP, bar_index, tpPrice)
            line.set_xy2(lineTP, bar_index + 1, tpPrice)

// ============================================
// VISUALISATION CONDITIONS SETUP / ENTRIES
// ============================================

// Setup triangles (signal UT Bot + touche OCC)
plotshape(longSetupSignal and not positionActive and not inBlackout and not forceClose, title="Setup L", style=shape.triangleup, location=location.belowbar, color=color.blue, size=size.small)
plotshape(shortSetupSignal and not positionActive and not inBlackout and not forceClose, title="Setup S", style=shape.triangledown, location=location.abovebar, color=color.blue, size=size.small)

// Entrées effectives (petites boules)
isNewLong  = positionActive and currentSide == 1 and not positionActive[1]
isNewShort = positionActive and currentSide == -1 and not positionActive[1]

plotshape(isNewLong, title="Entry L", style=shape.circle, location=location.belowbar, color=color.green, size=size.tiny)
plotshape(isNewShort, title="Entry S", style=shape.circle, location=location.abovebar, color=color.red, size=size.tiny)

// ============================================
// VISUALISATION SORTIES (croix bien visibles)
// ============================================

// SL (croix rouge)
plotshape(showSLLong, title="SL Long", style=shape.xcross, location=location.belowbar, color=color.red, size=size.tiny)
plotshape(showSLShort, title="SL Short", style=shape.xcross, location=location.abovebar, color=color.red, size=size.tiny)

// BE (croix orange)
plotshape(showBELong, title="BE Long", style=shape.xcross, location=location.belowbar, color=color.orange, size=size.tiny)
plotshape(showBEShort, title="BE Short", style=shape.xcross, location=location.abovebar, color=color.orange, size=size.tiny)

// TP Partiel (croix verte)
plotshape(showTPLong, title="TP Partiel Long", style=shape.xcross, location=location.abovebar, color=color.green, size=size.tiny)
plotshape(showTPShort, title="TP Partiel Short", style=shape.xcross, location=location.belowbar, color=color.green, size=size.tiny)

// TP Final (croix verte)
plotshape(showFinalLong, title="TP Final Long", style=shape.xcross, location=location.abovebar, color=color.green, size=size.tiny)
plotshape(showFinalShort, title="TP Final Short", style=shape.xcross, location=location.belowbar, color=color.green, size=size.tiny)

// Force Close 22h (croix orange)
plotshape(showForceLong, title="Force Close L", style=shape.xcross, location=location.abovebar, color=color.orange, size=size.tiny)
plotshape(showForceShort, title="Force Close S", style=shape.xcross, location=location.belowbar, color=color.orange, size=size.tiny)

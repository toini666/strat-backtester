roundToTick(float price) =>
    math.round(price / syminfo.mintick) * syminfo.mintick

getTimeframe() =>
    string tf = timeframe.period
    float tfNum = str.tonumber(tf)
    string result = if not na(tfNum)
        int tfMinutes = int(tfNum)
        tfMinutes >= 60 and tfMinutes % 60 == 0 ? "H" + str.tostring(tfMinutes / 60) : "M" + str.tostring(tfMinutes)
    else if str.endswith(tf, "S")
        "S" + str.substring(tf, 0, str.length(tf) - 1)
    else if tf == "D"
        "D1"
    else if str.endswith(tf, "D")
        "D" + str.substring(tf, 0, str.length(tf) - 1)
    else if tf == "W"
        "W1"
    else if str.endswith(tf, "W")
        "W" + str.substring(tf, 0, str.length(tf) - 1)
    else if tf == "M"
        "MO1"
    else if str.endswith(tf, "M")
        "MO" + str.substring(tf, 0, str.length(tf) - 1)
    else
        tf
    result

isInTimeSlot(int hh, int mm, int startH, int startM, int endH, int endM) =>
    int curMin   = hh * 60 + mm
    int startMin = startH * 60 + startM
    int endMin   = endH * 60 + endM
    bool endIsDayLastMinute = endH == 23 and endM == 59
    if startMin <= endMin
        curMin >= startMin and (endIsDayLastMinute ? curMin <= endMin : curMin < endMin)
    else
        curMin >= startMin or (endIsDayLastMinute ? curMin <= endMin : curMin < endMin)

isBlackoutWindow() =>
    int hh = hour(time_close, "Europe/Brussels")
    int mm = minute(time_close, "Europe/Brussels")
    bool b1 = i_bo1On and isInTimeSlot(hh, mm, i_bo1StartH, i_bo1StartM, i_bo1EndH, i_bo1EndM)
    bool b2 = i_bo2On and isInTimeSlot(hh, mm, i_bo2StartH, i_bo2StartM, i_bo2EndH, i_bo2EndM)
    bool b3 = i_bo3On and isInTimeSlot(hh, mm, i_bo3StartH, i_bo3StartM, i_bo3EndH, i_bo3EndM)
    bool b4 = i_bo4On and isInTimeSlot(hh, mm, i_bo4StartH, i_bo4StartM, i_bo4EndH, i_bo4EndM)
    bool b5 = i_bo5On and isInTimeSlot(hh, mm, i_bo5StartH, i_bo5StartM, i_bo5EndH, i_bo5EndM)
    bool b6 = i_bo6On and isInTimeSlot(hh, mm, i_bo6StartH, i_bo6StartM, i_bo6EndH, i_bo6EndM)
    b1 or b2 or b3 or b4 or b5 or b6

isAutoCloseBar() =>
    int openMin   = hour(time, "Europe/Brussels") * 60 + minute(time, "Europe/Brussels")
    int closeMin  = hour(time_close, "Europe/Brussels") * 60 + minute(time_close, "Europe/Brussels")
    int targetMin = i_autoCloseH * 60 + i_autoCloseM
    closeMin >= openMin ? (targetMin >= openMin and targetMin <= closeMin) : (targetMin >= openMin or targetMin <= closeMin)

method st(simple string src, float oscVal, simple int len) =>
    float o = switch src
        "SMA" => ta.sma(oscVal, len)
        "EMA" => ta.ema(oscVal, len)
    o

float hma2    = emaMain + (rawHma2 - emaMain) * i_ampMult

bool  canalGreen = hma1 > hma2
float canalUpper = math.max(hma1, hma2)
float canalLower = math.min(hma1, hma2)

calcOscillator() =>
    float hi = ta.highest(mL)
    float lo = ta.lowest(mL)
    float av = ta.sma(hl2, mL)
    osc.sig := ta.ema(ta.linreg((close - math.avg(hi, lo, av)) / (hi - lo) * 100, mL, 0), sL)
    osc.sgD := sT.st(osc.sig, 2)

calcMFI() =>
    mf.mfi := currentMfi

updateMFITracking() =>
    if mf.mfi > 0
        if mf.brT.size() > 1
            mf.brT.pop()
        if mf.blT.size() > mfL
            mf.blT.pop()
        if mf.mfi > mf.blT.avg()
            mf.blT.unshift(mf.mfi)
        else
            float histMfi = currentMfi[mfL]
            mf.blT.unshift(not na(histMfi) and histMfi > 0 ? histMfi : mf.mfi)
    if mf.mfi < 0
        if mf.blT.size() > 1
            mf.blT.pop()
        if mf.brT.size() > mfL
            mf.brT.pop()
        if mf.mfi < mf.blT.avg()
            mf.brT.unshift(mf.mfi)
        else
            float histMfi = currentMfi[mfL]
            mf.brT.unshift(not na(histMfi) and histMfi < 0 ? histMfi : mf.mfi)

calcOscillator()

bool hwExtremeShortOk = not i_hwExtremeOn or na(lastConfirmedHWValue) or lastConfirmedHWValue >= -i_hwExtreme

// ③ Extrêmes SIG courant
bool sigExtremeLongOk  = not i_sigExtremeOn or na(oscSig) or oscSig <= i_hwExtreme
bool sigExtremeShortOk = not i_sigExtremeOn or na(oscSig) or oscSig >= -i_hwExtreme

// ④ Range HW
bool hwRangeOk = not i_hwRangeOn or na(lastConfirmedHWValue) or math.abs(lastConfirmedHWValue) > i_hwRange

// ⑤ Nuage MFI

bool  cloudLongOk       = hasActiveMfiCloud and activeMfiRefLine < activeMfiCloud
bool  cloudShortOk      = hasActiveMfiCloud and activeMfiRefLine > activeMfiCloud

// ⑥ Deltas
bool deltaLongOn  = oscSig > 0 and mf.mfi > 0
bool deltaShortOn = oscSig < 0 and mf.mfi < 0

bool deltaShortOk = not i_deltaOn or not deltaLongOn

// ⑦ Nuage < 0 / > 0
bool cloudZeroLongOk  = not i_cloudZeroOn or mf.mfi < 0
bool cloudZeroShortOk = not i_cloudZeroOn or mf.mfi > 0

// ⑧ Extinction deltas
bool deltaExtLongOk  = not i_deltaExtOn or (deltaShortOn[1] and not deltaShortOn)
bool deltaExtShortOk = not i_deltaExtOn or (deltaLongOn[1]  and not deltaLongOn)

// ⑤+⑥ Gate combiné (OR)
bool cloudOrDeltaLongOk  = not i_cloudOn or cloudLongOk  or (i_deltaOn and deltaLongOn)
bool cloudOrDeltaShortOk = not i_cloudOn or cloudShortOk or (i_deltaOn and deltaShortOn)

// Résultat oscillateur global
bool oscAllLongOk  = hwDirLongOk  and hwExtremeLongOk  and sigExtremeLongOk  and hwRangeOk and cloudOrDeltaLongOk  and deltaLongOk  and cloudZeroLongOk  and deltaExtLongOk
bool oscAllShortOk = hwDirShortOk and hwExtremeShortOk and sigExtremeShortOk and hwRangeOk and cloudOrDeltaShortOk and deltaShortOk and cloudZeroShortOk and deltaExtShortOk

// Long: bougie haussière qui ouvre sous l'EMA principale et clôture au-dessus
bool emaBreakLongRaw  = close > open and open < emaMain and close > emaMain
// Short: bougie baissière qui ouvre au-dessus de l'EMA principale et clôture en-dessous
bool emaBreakShortRaw = close < open and open > emaMain and close < emaMain

if emaBreakLongRaw
    barsSinceLongBreak     := 0
    lastBreakLongLow       := low
    lastBreakLongSizePct   := candlePct
    lastBreakLongSizeOk    := breakCandleSizeOk

    barsSinceShortBreak    := 0
    lastBreakShortHigh     := high
    lastBreakShortSizePct  := candlePct
    lastBreakShortSizeOk   := breakCandleSizeOk

bool storedShortBreakOk = barsSinceShortBreak <= i_breakWaitBars and lastBreakShortSizeOk

bool  slShortOk   = slDistShort <= i_maxSlPoints

if storedLongBreakOk and emaAboveSSL and canalGreen and oscAllLongOk and slLongOk and cooldownOk
    longEntry          := true
    barsSinceLongBreak := 9999

if storedShortBreakOk and emaBelowSSL and not canalGreen and oscAllShortOk and slShortOk and cooldownOk
    shortEntry          := true
    barsSinceShortBreak := 9999

// Calcul TP/SL
float calcEntryLong     = close

float calcStopShort     = roundToTick(slShort)

    forceCloseTriggered := true
    forceCloseSide      := currentSide
    forceCloseInProfit  := currentSide == 1 ? close > entryPrice : close < entryPrice
    positionActive      := false
    currentSide         := 0
    lastActionBar       := bar_index
    lastCloseBarIndex   := bar_index
    tpPartialHit        := false
    hmaInTpHit          := false
    canAct              := false
    if not na(lineEntry)
        line.delete(lineEntry)
        lineEntry := na
    if not na(lineStop)
        line.delete(lineStop)
        lineStop := na
    if not na(lineTP)
        line.delete(lineTP)
